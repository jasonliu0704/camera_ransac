def drawlines(img1,img2,lines,pts1,pts2):
    ''' img1 - image on which we draw the epilines for the points in img2
        lines - corresponding epilines '''
    r,c = img1.shape
    img1 = cv2.cvtColor(img1,cv2.COLOR_GRAY2BGR)
    img2 = cv2.cvtColor(img2,cv2.COLOR_GRAY2BGR)
    for r,pt1,pt2 in zip(lines,pts1,pts2):
        color = tuple(np.random.randint(0,255,3).tolist())
        x0,y0 = map(int, [0, -r[2]/r[1] ])
        x1,y1 = map(int, [c, -(r[2]+r[0]*c)/r[1] ])
        img1 = cv2.line(img1, (x0,y0), (x1,y1), color,1)
        img1 = cv2.circle(img1,tuple(pt1),5,color,-1)
        img2 = cv2.circle(img2,tuple(pt2),5,color,-1)
    return img1,img2

import random

def ransac():
    threshold = 0.001
    N = 1000
    inlier_g = 0
    for i in range(0,N):
        # get 8 samples
        rand_matches = random.sample(good, 8)
        p = []
        q = []
        for m in rand_matches:
            p.append(kp1[m[0].queryIdx].pt)
            q.append(kp2[m[0].trainIdx].pt)
        p = np.int32(p)
        q = np.int32(q)
        # calculate F
        A = []
        for k in range(0,8):
            u = p[k][0]
            v = p[k][1]
            up = q[k][0]
            vp = q[k][1]
            A.append([u*up, v*up, up, u*vp, v*vp, vp, u, v, 1])
        U, s, V = np.linalg.svd(A, full_matrices=True)
        F = np.transpose(V[:, 8].reshape((3, 3)))
        U, s, V = np.linalg.svd(F, full_matrices=True)
        F = U.dot(np.diag([s[0], s[1], 0])).dot(np.transpose(V))

        # inliner counting
        inlier_count = 0
        for k in range(0, len(pts1)):
            x1 = np.array([pts1[k][0], pts1[k][1], 1])
            x2 = np.array([pts2[k][0], pts2[k][1], 1])
            if(x1.dot(F).dot(np.transpose(x2)) < threshold):
                inlier_count +=1
        # compare previous model
        if(inlier_count > inlier_g):
            inlier_g = inlier_count
            F_g = F
    return F_g
